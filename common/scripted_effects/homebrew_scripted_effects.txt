hb_add_development_from_building = { #credit to the original authors, but imma hijack this to make my stuff easier
	if = {
		limit = {
			FROM = { 
				OR = {
					has_country_flag = improve_development_on_buildings_built 
					AND = {
						has_reform = northern_league_magnates
						faction_in_power = nl_cooperatists
					}
				}
			}
			NOT = { gained_development_from_building = { building = $building$ } }
		}
		add_base_$type$ = $value$
		hidden_effect = {
			set_province_flag = built_dev_$building$
		}
	}
}
hb_add_development_from_upgrading_or_building = {
	if = {
		limit = {
			FROM = { 
				OR = {
					has_country_flag = improve_development_on_buildings_built 
					AND = {
						has_reform = northern_league_magnates
						faction_in_power = nl_cooperatists
					}
				}
			}
			NOT = { gained_development_from_building = { building = $building$ } }
		}
		if = {
			limit = { has_province_flag = built_dev_$obsolete_building$ }
			add_base_$type$ = $upgrade_value$
		}
		else = {
			add_base_$type$ = $value$
		}
		hidden_effect = {
			set_province_flag = built_dev_$building$
		}
	}
}
hb_remove_development_from_building = {
	if = {
		limit = {
			FROM = {
				OR = {
					has_country_flag = improve_development_on_buildings_built 
					has_reform = northern_league_magnates
				}
			}
			gained_development_from_building = { building = $building$ }
		}
		add_base_$type$ = -$value$
		if = {
			limit = {
				NOT = { base_$type$ = 1 }
			}
			set_base_$type$ = 1
		}
		hidden_effect = {
			clr_province_flag = built_dev_$building$
		}
	}
}
hb_add_melting_pot = {
	if = { 
		limit = { 
			owner = { 
				has_idea = homebrew_melting_pot 
				has_country_flag = hb_has_melting_pot
				OR = {
					high_tolerance_$race$_race_trigger = yes
					medium_tolerance_$race$_race_trigger = yes
				}
				NOT = { has_country_modifier = $race$_administration } 
			}
		}
		add_province_modifier = { name = meltingpot_$race$ duration = -1 hidden = yes }
	}
}
hb_remove_melting_pot = {
	if = { 
		limit = { 
			NOT = { has_any_$race$_pop_trigger = yes } 
		}
		remove_province_modifier = meltingpot_$race$
	}
}
homebrew_liquidation = { #takes a value arguement
	if = { limit = { FROM = { has_idea = kriegsproduktion3 } }
		custom_tooltip = homebrew_liquidation_tt
		hidden_effect = {
			set_variable = { which = build_cost_payout value = $value$ } #converts build cost into a variable (can build cost be directly exported to this function? It would remove the need for an arguement)
			set_variable = { which = base_build_cost value = 100 } #represents 100%
			export_to_variable = { which = owner_build_cost value = modifier:build_cost who = FROM }
			export_to_variable = { which = province_build_cost value = modifier:local_build_cost who = ROOT }
			change_variable = { which = owner_build_cost which = province_build_cost } #totals the two variables above to represent the provincial build cost reduction
			multiply_variable = { which = owner_build_cost value = 100 } #multiplies above to make it a %
			change_variable = { which = base_build_cost which = owner_build_cost } #substract the base by the reduction to a) filter through the if check and b) set a variable to be used to alter the payout variable
			if = { limit = { check_variable = { which = base_build_cost value = 33 } } #so long as the local build cost is not under 33%
				multiply_variable = { which = build_cost_payout value = 0.33 }
			}
			else = {
				multiply_variable = { which = build_cost_payout which = base_build_cost }
				divide_variable = { which = build_cost_payout value = 100 }
			}
			while = { limit = { check_variable = { which = build_cost_payout value = 1 } } #basic while loop, might upgrade to binary
				FROM = { add_treasury = 1 }
				change_variable = { which = build_cost_payout value = -1 }
			}
		}
	}
}
add_central_planning = { #takes a level arguement
	clear_metropolis_planning = yes
	add_province_modifier = { name = metropolis_planning_$level$ duration = -1 hidden = yes }
	if = { limit = { OR = { has_terrain = dwarven_hold has_terrain = dwarven_hold_surface } }
		clear_hold_ei = yes
		add_province_modifier = { name = hold_ei_$level$ duration = -1 hidden = yes }
	}
}
